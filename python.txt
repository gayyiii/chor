# 1 Defining a class named 'Student'

class Student:
    # Constructor (automatically called when an object is created)
    def __init__(self, name, roll_no, marks):
        self.name = name
        self.roll_no = roll_no
        self.marks = marks

    # Method to display student details
    def display_details(self):
        print("Student Name:", self.name)
        print("Student Roll No:", self.roll_no)
        print("Student Marks:", self.marks)
        print()

    # Method to calculate grade based on marks
    def calculate_grade(self):
        if self.marks >= 75:
            return "A"
        elif self.marks >= 60:
            return "B"
        elif self.marks >= 40:
            return "C"
        else:
            return "Fail"


# Creating objects of the class Student
student1 = Student("John", 101, 85)
student2 = Student("Sara", 102, 67)
student3 = Student("Tom", 103, 35)

# Displaying student details
print("======== Student Details =========")
student1.display_details()
student2.display_details()
student3.display_details()

# Displaying grades for each student
print("======== Student Grades =========")
print(student1.name, "Grade:", student1.calculate_grade())
print(student2.name, "Grade:", student2.calculate_grade())
print(student3.name, "Grade:", student3.calculate_grade())

import array

# 1. Creating an array of integers
numbers = array.array('i', [10, 20, 30, 40, 50])
print("Original Array:", numbers.tolist())   # tolist() displays it like a normal list
print()

# 2. Accessing elements using index
print("First element:", numbers[0])
print("Last element:", numbers[-1])
print()

# 3. Adding elements to array
numbers.append(60)                 # Adds one element at end
numbers.extend([70, 80])           # Adds multiple elements
print("After adding elements:", numbers.tolist())
print()

# 4. Inserting an element at specific position
numbers.insert(2, 25)              # Insert 25 at index 2
print("After inserting 25 at index 2:", numbers.tolist())
print()

# 5. Removing elements
numbers.remove(40)                 # Removes first occurrence of 40
print("After removing 40:", numbers.tolist())
print()

# 6. Popping last element
numbers.pop()                      # Removes last element
print("After popping last element:", numbers.tolist())
print()

# 7. Iterating through array
print("Elements in array:")
for num in numbers:
    print(num, end=" ")
print("\n")

print("Sum of array elements:", sum(numbers))
print()
